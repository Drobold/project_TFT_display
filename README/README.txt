#define GPIO_Pin_CS GPIO_Pin_2
#define GPIO_Pin_DC GPIO_Pin_3
#define GPIO_Pin_RES GPIO_Pin_4
#define GPIO_Pin_SCL GPIO_Pin_5
#define GPIO_Pin_BLK GPIO_Pin_6
#define GPIO_Pin_SDA GPIO_Pin_7
//*********************************************************************************************************//
ПОДКЛЮЧЕНИЕ КОНТАКТОВ ДИСПЛЕЯ К STM32F103C8T6
1. GND (Земля)

2. VCC (Питание)

3. SCL - GPIOA, PIN 5

4. SDA - GPIOA, PIN 7

5. RESET - GPIOA, PIN 4

6. DC - GPIOA, PIN 3

7. CS - GPIOA, PIN 2

8.BLK - GPIOA, PIN 6
//*********************************************************************************************************//




//********************************//
ОЧЕНЬ ВАЖНО!!!
80-(X) 160-(Y). Начало системы отсчета начинается с левого верхнего угла.
Заполнение пикселей идет сверху вниз, слева на право

Диаппазон видимых пикселей, учитывая смещение по оси X относительно системы отсчета дисплея
X(26;105)
